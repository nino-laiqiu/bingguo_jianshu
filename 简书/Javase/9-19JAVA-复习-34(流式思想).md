##stream流中常用的一些方法

 stream的特点://stream流是管道流 只能被消费(执行一次)

```
    //stream流是管道流 只能被消费(执行一次)
    public static void main(String[] args) {
        //foreach方法  Consumer接口(消费)
        //终止函数 遍历之后其他函数不能执行
        Stream<Integer> stream1 = Stream.of(1, 2, 3, 4, 5, 6);
        stream1.forEach(name -> System.out.println(name));

        //filter过滤函数  Predicate接口(判断)
        Stream<String> stream2 = Stream.of("中国", "中华", "中原", "大陆");
        //返回另一个stream对象
        Stream<String> stream = stream2.filter(name -> name.startsWith("中"));
        stream.forEach(name -> System.out.println(name));

        //map方法 映射   Function接口(转换)   注意这里的转换不只是基本数据类型还可以是引用类型 比如自定义的people类型  注意一下
        Stream<String> streammap = Stream.of("11", "22", "33", "44");
        Stream<Integer> stream4map = streammap.map(name -> Integer.parseInt(name));
        stream4map.forEach(name -> System.out.println(name));

        //count统计个数 终结方法
        Stream<Integer> streamcount = Stream.of(1, 2, 3, 4, 5, 5, 6, 7, 8, 9);
        System.out.println(streamcount.count());

        //limit 截取 这是一个延迟方法
        Stream<Integer> stream3 = Stream.of(1, 2, 3, 4, 5, 6);
        Stream<Integer> limit = stream3.limit(2);
        limit.forEach(name -> System.out.println(name));

        //skip 跳过 假设不足则显示0
        Stream<Integer> stream4 = Stream.of(1, 2, 3, 4, 5, 6);
        Stream<Integer> skip = stream4.skip(6);
        skip.forEach(name -> System.out.println(name));


        //concat 合并流
        Stream<Integer> s1 = Stream.of(1, 2, 3, 4, 5);
        Stream<Integer> s2 = Stream.of(1, 2, 3, 4, 5);
        Stream<Integer> concat = Stream.concat(s1, s2);
        concat.forEach(name-> System.out.println(name));

    }
}
```

#####2.方法:筛选与切片(filter limit skip distinct)
```
    //帅选与切片
    public static void main(String[] args) {
        List<String> listStream = new ArrayList<>();
        listStream.add("1");
        listStream.add("1");
        listStream.add("2");
        listStream.add("3");
        listStream.add("4");
        listStream.add("5");

        //排除方法
        listStream.stream().filter(number->number.equals("1")).forEach(System.out::println);
        //截断方法
        listStream.stream().limit(2).forEach(System.out::println);
        //跳过方法
        listStream.stream().skip(2).forEach(System.out::println);
        //筛选去重
        listStream.stream().distinct().forEach(System.out::println);
    }
}
```

```
 //匹配与查找
    public static void main(String[] args) {
        List<People> peopleSteam = new ArrayList<>();
        peopleSteam.add(new People("张无忌", "18", 4000));
        peopleSteam.add(new People("周芷诺", "19", 9000));
        peopleSteam.add(new People("张三丰", "1", 66666));
        peopleSteam.add(new People("赵敏", "10", 45678));
        peopleSteam.add(new People("小明", "66", 9780));

        //allMatch方法 判断是否都
        boolean allMatch = peopleSteam.stream().allMatch(people -> people.getAge().equals("18"));
        System.out.println(allMatch);//false

        //anyMatch方法 判断是否有一个满足
        boolean anyMatch = peopleSteam.stream().anyMatch(people -> people.getSelarly() > 66665);
        System.out.println(anyMatch);//ture

        //noneMatch方法  判断是否没有匹配元素 没有则返回ture
        boolean noneMatch = peopleSteam.stream().noneMatch(people -> people.getName().equals("张无忌1"));
        System.out.println(noneMatch);//ture

        //findFirst方法 返回第一个元素
        Optional<People> first = peopleSteam.stream().findFirst();
        System.out.println(first);//Optional[People{name='张无忌', age='18', selarly=4000}]

        //findAny方法 返回任意一个元素
        Optional<People> any = peopleSteam.stream().findAny();
        System.out.println(any);//Optional[People{name='张无忌', age='18', selarly=4000}]

        //count方法 返回元素的总数
        System.out.println(peopleSteam.stream().count());//5

        //max方法 返回最大的 min同理返回最小的 锥字符串的比较方法
        Optional<People> max = peopleSteam.stream().max((s1, s2) -> (int) (s1.getSelarly() - s2.getSelarly()));
        System.out.println(max);//Optional[People{name='张三丰', age='1', selarly=66666}]

        Optional<People> max1 = peopleSteam.stream().max((s1, s2) -> Long.compare(s1.getSelarly(), s2.getSelarly()));
        System.out.println(max1);//Optional[People{name='张三丰', age='1', selarly=66666}]

        Optional<People> max2 = peopleSteam.stream().max((s1, s2) -> s1.getName().compareTo(s2.getName()));
        System.out.println(max2);//Optional[People{name='赵敏', age='10', selarly=45678}]

        Optional<String> max3 = peopleSteam.stream().map(people -> people.getName()).max(String::compareTo);
        System.out.println(max3);//Optional[赵敏]

        //foreach方法 内部迭代
        peopleSteam.stream().forEach(System.out::println);
    }
}
```
