public boolean canRead()                  是否可读
public boolean canWrite()                 是否可写

public String[] list(FilenameFilter filter)                                        获取所有子文件名称

package Day14;

import java.io.File;
import java.io.FilenameFilter;

public class Demo3 {
    public static void main(String[] args) {
//getfiles("C:/picture",".jpg");
        getfile2("C:/picture");
    }
    public  static void getfile(String path, String low){
        File file = new File(path);
        File[] files = file.listFiles();
        for (File file1 : files) {
            if (file1.isFile() && file1.getName().toUpperCase().endsWith(low.toUpperCase()))
            {
                System.out.println(file1.getAbsolutePath());
            }
        }

    }
    public  static  void getfiles(String path,String wow){
        //封装
        File file = new File(path);
        //这是一个方法 返回抽象路径名数组，这些路径名表示此抽象路径名表示的目录中满足指定过滤器的文件和目录
        File[] files = file.listFiles(new FilenameFilter() {
            @Override
            public boolean accept(File file, String s) {
                File file1 = new File(file, s);
                if (file1.isFile() && file1.getName().toUpperCase().endsWith(wow.toUpperCase())) {
                    return true;
                }
                return false;
            }
        });
        for (File file1 : files) {

            System.out.println(file1.getAbsolutePath());
        }

    }
    //递归删除
    public static void getfile2(String path){
        File file = new File(path);
        if (file.isFile()){
            file.delete();
        }
        else{
            File[] files = file.listFiles();
            if (files.length >0 && files!= null){
                for (File file1 : files) {
                    getfile2(file1.getAbsolutePath());
                }
            }
            file.delete();
        }
    }
    //递归遍历目录下指定后缀名结尾的文件名称
    public  static  void getfiles11(String path,String wow){
        File file = new File(path);
        if (file.isFile()){
            if (file.getName().toUpperCase().endsWith(wow)){
                System.out.println(file.getAbsolutePath());
            }
        }
        else {
            File[] files = file.listFiles();
            if (files.length > 0 && files!=null){
                for (File file1 : files) {
                    getfiles11(file1.getAbsolutePath(),wow);
                }
            }
        }

    }
}
