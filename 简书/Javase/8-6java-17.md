2.set接口
```
mport java.util.HashSet;
import java.util.Iterator;
import java.util.Set;

public class Demo16 {
    public static void main(String[] args) {
        Set<Integer> set = new HashSet<>();
        set.add(1);
        set.add(9);
        set.add(12);
        System.out.println(set);
//增强循环
        for (Integer integer : set) {
             System.out.println(integer);
//迭代器循环
            Iterator<Integer> iterator = set.iterator();
            while (iterator.hasNext()) {
                Integer next = iterator.next();
                System.out.println(next);
            }
        }
    }
}
```

```
import java.util.HashMap;
import java.util.Set;

public class Demo17 {
    public static void main(String[] args) {
        HashMap<String, Integer> objectObjectHashMap = new HashMap<>();
        objectObjectHashMap.put("小明", 199);
        objectObjectHashMap.put("小华", 167);
        objectObjectHashMap.put("小额", 166);
        Set<String> strings = objectObjectHashMap.keySet();

        for (String key : strings) {
            Integer integer = objectObjectHashMap.get(key);
            System.out.println(integer + "=" + key);
        }
        
    }
}
```
