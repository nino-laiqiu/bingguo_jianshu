##1.生产者消费者问题
![image.png](https://upload-images.jianshu.io/upload_images/9049859-a755a8fbc676ea54.png?imageMogr2/auto-orient/strip%7CimageView2/2/w/1240)
```
public class Test5 {
    public static void main(String[] args) {
        Check check = new Check();
        Consumer consumer = new Consumer(check);
        Producer producer = new Producer(check);
        Thread consumerthread = new Thread(consumer);
        Thread consumerthread3 = new Thread(consumer);
        Thread producerthread1 = new Thread(producer);
        consumerthread.setName("消费者一");
        consumerthread3.setName("消费者二");
        producerthread1.setName("生产者一");
        consumerthread.start();
        consumerthread3.start();
        producerthread1.start();


    }

}

class Check {
    private int productnum = 0;

    public synchronized void consumproduction() {
        if (productnum > 0) {
            productnum--;
            System.out.println(Thread.currentThread().getName() + "消费一个" + "还剩" + productnum);
            notify();
        } else {
            try {
                wait();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }

    public synchronized void producerProduction() {
        if (productnum < 20) {
            productnum++;
            System.out.println(Thread.currentThread().getName() + "生产一个" + "还剩" + productnum);
            notify();

        } else {
            try {
                wait();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

class Consumer implements Runnable {
    private Check check;

    public Consumer(Check check) {
        this.check = check;
    }

    @Override
    public void run() {
        while (true) {
            try {
                Thread.sleep(1000);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            check.consumproduction();
        }
    }
}

class Producer implements Runnable {
    private Check check;

    public Producer(Check check) {
        this.check = check;
    }

    @Override
    public void run() {
        while (true) {
            try {
                Thread.sleep(700);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            check.producerProduction();
        }
    }
}
```
