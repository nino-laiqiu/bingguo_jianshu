```
   //方法引用
    //使用lambda表达式
public static void main(String[] args) {
    List<String> objects = new ArrayList<>();
    objects.add("貂蝉");
    objects.add("董卓");
    objects.add("吕布");

     //lambda表达式
    //objects.stream().forEach(s -> System.out.println(objects));
    objects.forEach(s-> System.out.println(s));
    //简化版 方法引用
   // objects.stream().forEach(System.out::println);
    objects.forEach(System.out::println);



    //对四种接口的使用

    //对象引用::实例方法名
    //Consumer<String> consumer = name-> System.out.println(name);
    Consumer<String> consumer1 =System.out::println;
    consumer1.accept("hello world");


    //类名::静态方法名
   // Function<Integer,Integer> function =number->Math.abs(number);
    Function<Integer,Integer> function1 = Math::abs;
    System.out.println(function1.apply(-99));


    //类名::实例方法名
   /* Predicate<String> predicate =s -> s.length()>5;
    System.out.println(predicate.test("12345"));*/

    BiPredicate<String,String> biPredicate = (s, s2) -> s.startsWith(s2);
    BiPredicate<String,String> biPredicate1 = String::startsWith;
    System.out.println(biPredicate1.test("中国", "中"));

    //引用构造器
/*
    Function<Integer,String[]> function = s -> new String[s];
    System.out.println(Arrays.toString(function.apply(12)));*/
   // Function<Integer,StringBuffer> function2 = s->new StringBuffer(s);
    Function<Integer,StringBuffer> function3 = StringBuffer::new;
    StringBuffer apply = function3.apply(10);
    //创建一个初始容量大小为 10 的stringbuff
    System.out.println(apply);


    //引用数组
    Function<Integer,String[]> function = s -> new String[s];
    Function<Integer,String[]> function5 = String[]::new;
    //创建初始容量大小为10的数组
    System.out.println(Arrays.toString(function5.apply(10)));
}
}
```
