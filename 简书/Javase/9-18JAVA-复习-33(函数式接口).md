2.函数式接口作为方法的案例
```
public class LambdaTest {
    //开启线程
    public  static void Thead1(Runnable runnable){
        new Thread(runnable).start();
    }

    public static void main(String[] args) {
        Thead1(new Runnable() {
            @Override
            public void run() {
                System.out.println("开启多线程 --->" + Thread.currentThread().getName() );
            }
        });
        //函数式接口  简化
        Thead1(()-> System.out.println("开启多线程 --->" + Thread.currentThread().getName())
        );
    }
}
```

Lambda表达式传参案例

```
public class LambdaTest {
    //定义一个比较方法
    public  static Comparator<String> getComparator(){
        return new Comparator<String>() {
            @Override
            public int compare(String s, String t1) {

                  return  t1.length()-s.length();
            }
        };
    }
    //采用lambda表达式及简化版
    public  static  Comparator<String>  getComparatorlambda(){
          return  (String s ,String t1)->{
              return t1.length()-s.length();
          };
    };
    //简化版
    public  static  Comparator<String>  getComparatorlambda1(){
        return  (s,t1)->t1.length()-s.length();
    }

    public static void main(String[] args) {
        //建立一个数组
        String[] strings = {"222","44","1111","355555"};
        //排序之前
        System.out.println(Arrays.toString(strings));
        //使用arrays的排序方法
         Arrays.sort(strings,getComparator());
        Arrays.sort(strings,getComparatorlambda());
        Arrays.sort(strings,getComparatorlambda1());
        System.out.println(Arrays.toString(strings));
    }
}
```
(lambda输出只有一行的情况下可以省略{} 和 ";")

##常用接口-Consumer 消费

```
public class LambdaTest {
    //定义一个方法,消费字符串
    public  static void method(String s,Consumer<String> consumer1,Consumer<String> consumer2) {
        consumer1.accept(s);
        consumer2.accept(s);
       // Consumer中有连接方法,连接的位置不同哪个就先执行
        consumer1.andThen(consumer2).accept(s);

    }
     //lambda写法
    public static void main(String[] args) {
        method("年糕",(string)->{
                    System.out.println(string.toUpperCase() + System.currentTimeMillis());
                },
                (string)->{
                    System.out.println(string.toLowerCase() + System.currentTimeMillis());
                });
    }
}
```

public class LambdaTest {
    //定义一个方法  测试某个字符串是否长度大于3以及是否包含字符串"9"
    public  static boolean method(String string ,Predicate<String> predicate1,Predicate<String> predicate2){
      //  return  predicate1.test(string) && predicate2.test(string);
       return predicate1.and(predicate2).test(string);
    }
    //定义一个方法 测试某个字符串是否长度大于3或者包含字符串"3"
    public  static  boolean method1(String string,Predicate<String> predicate1,Predicate<String> predicate2){
       // return  predicate1.test(string) || predicate2.test(string);
        return  predicate1.or(predicate2).test(string);
    }
    //定义一个方法 测试某个字符串长度是否大于5
    public static  boolean method3(String string ,Predicate<String> predicate){
       // return  !predicate.test(string);
        return  predicate.negate().test(string);
    }


    public static void main(String[] args) {
        String string ="123456";
        boolean method = method(string, s -> s.length() > 3, s -> s.contains("9"));
        System.out.println(method);

        boolean method1 = method(string, s -> s.length() > 3, s -> s.contains("3"));
        System.out.println(method1);


        boolean b = method3(string, s -> s.length() > 3);
        System.out.println(b);
    }
}
