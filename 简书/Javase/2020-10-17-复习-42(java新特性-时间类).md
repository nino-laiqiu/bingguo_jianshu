#####解决多线程问题:(DateTimeFormatter  LocalDate  )
```
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.*;

public class Time_Test {
    public static void main(String[] args) throws Exception {
        DateTimeFormatter dtf = DateTimeFormatter.ofPattern("yyyyMMdd");
        Callable<LocalDate> task = new Callable<LocalDate>() {
            @Override
            public LocalDate call() throws Exception {
                return LocalDate.parse("20201017", dtf);
            }
        };
        ExecutorService poll = Executors.newFixedThreadPool(10);
        List<Future<LocalDate>> list = new ArrayList<>();
        for (int i = 0; i < 10; i++) {
               list.add(poll.submit(task));
        }
        for (Future<LocalDate> localDateFuture : list) {
            System.out.println(localDateFuture.get());
        }
    }
}
```

import java.time.*;
import java.time.format.DateTimeFormatter;

public class Java_Time {
    public static void main(String[] args) {

    }
    //LocalDate  localTime LocalDateTime
    @Test
    public void time(){
        System.out.println(LocalDate.now());
        System.out.println(LocalDate.now().format(DateTimeFormatter.ISO_DATE));
        System.out.println(LocalDate.of(2020, 12, 1));
    }
    //时间戳
    @Test
    public  void time1(){
        /*2020-10-17T12:32:37.593791800Z
        2020-10-17T20:32:37.598778300+08:00
        1602937957599*/
        System.out.println(Instant.now());//UTC时区
        System.out.println(Instant.now().atOffset(ZoneOffset.ofHours(8)));
        System.out.println(Instant.now().toEpochMilli());
    }
    //时间之间的间隔
    @Test
    public  void time2(){
        Instant now = Instant.now();
        try {
            Thread.sleep(1000);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        Instant now1 = Instant.now();
        Duration time = Duration.between(now, now1);
        System.out.println(time.toMillis());
        System.out.println(time.getSeconds());
    }
    @Test
    public  void  time4(){
        LocalDate of = LocalDate.of(2020, 1, 1);
        LocalDate now = LocalDate.now();
        System.out.println(LocalDate.now());//2020-10-17
        System.out.println(LocalDateTime.now());//2020-10-17T20:47:12.956250400
        Period between = Period.between(of, now);
        System.out.println(between);
        //计算相差多少年 月 日
        System.out.println(between.getYears());
        System.out.println(between.getMonths());
        System.out.println(between.getDays());

    }
}
