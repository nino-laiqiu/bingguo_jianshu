package JDBCMysql;

import com.alibaba.druid.pool.DruidDataSourceFactory;

import javax.sql.DataSource;
import java.io.IOException;
import java.sql.Connection;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;
import java.util.Properties;

public class DruidLinker {
    //提升为成员变量
    static DataSource poll;
     //静态代码快一次加载
    static {
        Properties properties = new Properties();
        //加载配置文件
         try {
             properties.load(Thread.currentThread().getContextClassLoader().getResourceAsStream("druid.properties"));
         } catch (IOException e) {
             e.printStackTrace();
         }
         //德鲁伊池
         try {
             poll = DruidDataSourceFactory.createDataSource(properties);
         } catch (Exception e) {
             e.printStackTrace();
         }

     }
     //建立连接
   public   static Connection getconnection() throws SQLException {
        return  poll.getConnection();
    }
    //关闭流 注意对于Caonnection来说只要归还
    public  static void CloseIO(ResultSet resultSet, Statement statement,Connection connection) throws SQLException {
        if (resultSet!=null){
            resultSet.close();
            resultSet =null;
        }
        if (statement != null){
            statement.close();
            statement=null;
        }
        if (connection!=null){
            connection.close();
        }
    }
}
