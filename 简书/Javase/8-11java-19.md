public class Demo12 {
    public static void main(String[] args) {
        Demo9 demo9 = new Demo9();
        demo9.start();
        for (int i = 0; i < 50; i++) {
            System.out.println(i+10000);
        }
    }
}
class Demo9 extends Thread{
    @Override
    public void run() {
        for (int i = 0; i < 100; i++) {
            if (i%2 ==1){
                System.out.println(i);
            }
        }
    }
}
