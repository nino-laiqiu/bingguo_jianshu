8.昨天的一道题重新写了一遍
```
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;

public class Demo9 {
    public static void main(String[] args) {
        String st = "红孩儿,20;哪吒,10;木吒,10;貂蝉,1;红孩儿,30;红孩儿,10;貂蝉,2";
        HashSet<Pepole> objects = new HashSet<>();
        String[] split = st.split(";");
        for (int i = 0; i < split.length; i++) {
            String[] split1 = split[i].split(",");
            Pepole pepole = new Pepole(split1[0], Integer.parseInt(split1[1]));
            objects.add(pepole);
        }

        //求次数
        HashMap<String, Pepole> Map = new HashMap<>();
        for (int i = 0; i < split.length; i++) {
            String[] split1 = split[i].split(",");
            Pepole pepole = new Pepole(split1[0], Integer.parseInt(split1[1]), 1);
            if (!Map.containsKey(split1[0])) {
                Map.put(split1[0], pepole);
            } else {
                //获取一次
                Pepole pepole1 = Map.get(split1[0]);
                int age = pepole1.age;
                int count = pepole.count;
                age+=Integer.parseInt(split1[1]);
                count++;
                pepole1.age =age;
                pepole1.count =count;
                //重新赋值覆盖
                Map.put(split1[0],pepole1);

            }
        }
        System.out.println(Map);

        //使用迭代器遍历
        Iterator<Pepole> iterator = objects.iterator();
        while (iterator.hasNext()) {
            System.out.println(iterator.next());
        }
        //使用增强循环
        for (Pepole object : objects) {
            System.out.println(object);
        }
        //使用普通循环  不行
        for (int i = 0; i < objects.size(); i++) {
            System.out.println();
        }
    }
}

class Pepole {
    String name;
    int age;
    int count;

    public Pepole(String name, int age) {
        this.name = name;
        this.age = age;
    }

    @Override
    public String toString() {
        return "Pepole{" +
                "name='" + name + '\'' +
                ", age=" + age +
                ", count=" + count +
                '}';
    }

    public Pepole() {
    }

    public Pepole(String name, int age, int count) {
        this.name = name;
        this.age = age;
        this.count = count;
    }
}
```
