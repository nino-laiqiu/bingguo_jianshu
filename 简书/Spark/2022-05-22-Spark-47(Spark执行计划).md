package cn.sparksql.DAY9

import org.apache.spark.sql.SparkSession

object sqlParser {
  def main(args: Array[String]): Unit = {
    val spark = SparkSession.builder().master("local").getOrCreate()

    import spark.implicits._
    spark.createDataset(Seq((1,"小明",13921),(2,"小米",15821))).toDF("uid","name","phone").createTempView("uid_log")
    spark.createDataset(Seq((1,"100"),(2,"200"))).toDF("uid","order_id").createTempView("order_log")
    val sql =
      """
        |select
        |t1.uid,
        |t1.name,
        |t1.phone,
        |t2.order_id
        |from
        |(
        |select uid,name,phone from uid_log where uid is not null
        |) t1
        |join
        |(
        |select uid,order_id from order_log where uid is not null and order_id is not null
        |) t2
        |on t1.uid = t2.uid
        |where phone like "139%"
        |""".stripMargin
    spark.sql(sql).explain("extended")
    spark.close()

  }
}
