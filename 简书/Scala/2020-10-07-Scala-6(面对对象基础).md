class A(var iName: String, iAge: Int) {
  //属性   Scala提供了getset方法  加上注解   @BeanProperty
  @BeanProperty var name: String = iName
  var age: Int = iAge

  override def toString: String = {
    "A的属性" + this.name + this.age
  }

  //这是一个构造器
  def this(iName: String) {
    //要调用主构造方法,原因是来源于java的设计思想,假设不写就是调用父类的构造方法
    this(" ", 1)
    this.name = iName
  }

  //这是一个方法
  def A(iName: String): Unit = {
    this.name = iName
  }
}
