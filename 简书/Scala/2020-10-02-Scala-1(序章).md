##5.语言输出的三种方式(printf )
![image.png](https://upload-images.jianshu.io/upload_images/9049859-d53cdf19485d8bd4.png?imageMogr2/auto-orient/strip%7CimageView2/2/w/1240)
```
package MyScala

object Test1 {
  def main(args: Array[String]): Unit = {
    var name: String = "小明"
    var age: Int = 18
    var salary: Double = 1999.0
    var hight: Float = 190f
    println("姓名"+name +"\t\n" + "年龄"+age+"\t\n"+"薪水"+salary+"\t\n"+"高度"+hight )
    //格式化输出
    printf("姓名=%s 年龄=%d  薪水=%f 高度=%f\r\n",name,age,salary,hight)
    //姓名=小明 年龄=18  薪水=1999.000000 高度=190.000000

    printf("姓名=%s 年龄=%d  薪水=%.2f 高度=%.3f\r\n",name,age,salary,hight)
    //姓名=小明 年龄=18  薪水=1999.00 高度=190.000
    
    //支持$输入内容 编译器会解析
    println(s"个人信息如下\t\n姓名$name\t\n年龄$age\t\n薪水$salary\t\n高度$hight\t\n")
     //个人信息如下	
    //姓名小明	
    //年龄18	
    //薪水1999.0	
    //高度190.0	
    println(s"个人信息如下\t\n姓名$name\t\n年龄${age + 10 }\t\n薪水${salary + 10.000}\t\n高度$hight\t\n")
    //个人信息如下	
    //姓名小明	
    //年龄28	
    //薪水2009.0	
    //高度190.0	
  }
}
```
