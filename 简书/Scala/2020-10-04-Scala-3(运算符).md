#####1.算数运算符
注意一下:/ % 没有++ --
```
object Test4 {
  def main(args: Array[String]): Unit = {
       var num=10
    var num2=3
    var numdouble:Double=10
    var numdouble1:Double=10.0
    println(num/num2)//3
    println(numdouble/num2)//3.3333
     var  num5:Double=10.0/3
    println(num5)//???? 3.3333333333333335 和老师的不太一样
    println(numdouble1/num2)//3.3333
    println((numdouble1/num2).formatted("%.2f"))//3.33
    //%运算
    //a%b = a - (a/b)*b
    println(num%num2)//1
    println(10 % -3)//1
    println(-10 % -3)//-1
    //scala没有++  -- 使用+=1 -=1 代替
    var num6=10
    num6+=1
    println(num6)
  }
}
```
