#####7.fold函数 及 简写写法
```
   //flodreduce方法案例 及简化写法
    // foldright , foldleft fold 之间的区别同reduce
    var list = List(1, 2, 3, 4, 5)
    //等同于List(10,1,2,3,4,5)
    println(list.fold(10)(minus))
    //List(1,2,3,4,5,10)
    println(list.foldRight(10)(minus))
    //简写写法  (number/:集合)(函数) 或者 (number:\集合)(函数)
    println((10 /: list) (minus))
  }

  def minus(s1: Int, s2: Int) = {
    s1 + s2
  }
```
