object Test8 {
  def main(args: Array[String]): Unit = {
    //求最大值(递归)
    var list = List(1, 2, 3, 6, 7, 88)
    println(max(list))
    //反转
    var st = "HELLO"
    println(reverse(st))
    //阶乘
    var i: Int = 6
    println(factorial(i))

  }

  def max(list: List[Int]): Int = {
    if (list.isEmpty) {
      0
    }
    else if (list.size == 1) {
      list.head
    }
    else if (list.head > max(list.tail)) list.head
    else max(list.tail)

  }

  def reverse(string: String): String = {
    if (string.length == 1) string
    else reverse(string.tail) + string.head
  }

  def factorial(int: Int): Int = {
    println(int)
    if (int == 0) 1
    else int * factorial(int - 1) //犯了一个错误,以为为0,结果就是0,没有理解递归 先进后出
  }
}
