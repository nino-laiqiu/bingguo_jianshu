```
object Test1 {
  def main(args: Array[String]): Unit = {
    //对象匹配案例

    number match {

      case _ => println("none")
    }
  }
}
object Qquare {
  def unapply(arg: Double): Option[Double] = {
    println("执行unapply方法")
    Some(math.sqrt(arg))
  }
  //apply  分成一个实例
  //def apply(): Qquare = new Qquare()
  def apply(z:Double): Double  = z*z
}
```

```
    }
  }
}

object Quer1 {
  def unapplySeq(string: String): Option[Seq[String]] = {
    if (string.contains(","))
      Some(string.split(","))
    else None
  }
}
```
