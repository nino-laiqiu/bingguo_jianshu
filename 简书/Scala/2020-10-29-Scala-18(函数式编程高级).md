##5.匿名函数与高阶函数(嵌套匿名函数)
```
object Test {
  def main(args: Array[String]): Unit = {
    //类型推导-不需要返回值类型
    //= 写成 =>
    //(int: Int) => int * 2  这是一个匿名函数
    var s1 = (int: Int) => int * 2
    // println(s1)
    // println(s1(2))
    //函数只保证运行的正确性,不保证业务的正确性

    //复习高阶函数
    println(fun1(sum,11))
    //高阶函数的套用
    println(fun(sum, division, 7))
    //高阶函数返回一个匿名函数
    println(fun3(2)(4))
    //上面的写法等价于
    var d = fun3(2)
    println(d(4))
  }

  def fun3(int: Int)={
    (int1:Int)=>int+int1
  }
  def  fun1 (f:Int=>Double,int: Int)={
       f(int)
  }
  def fun(f: Int => Double, f1: Double => Double, int: Int) = {
    //套用
    f1(f(int))
  }

  def sum(int: Int): Double = {
    int.asInstanceOf[Double] + 1
  }

  def division(double: Double) = {
    double % 2
  }
}
```

##9.控制抽象
(用控制抽象来模拟while循环)
```
object Test7 {
  def main(args: Array[String]): Unit = {
    var x = 10
    //()就是fun方法  {}就是fun2方法
    until(x > 0) {
      x -= 1
      println(x)
    }
  }

  def until(fun: => Boolean)(fun2: => Unit): Unit = {
    if (fun) {
      fun2
      until(fun)(fun2)
    }
  }
}
```
