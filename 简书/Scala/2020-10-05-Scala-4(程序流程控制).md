object Test4 {
  def main(args: Array[String]): Unit = {
    var list =Seq("你好","i love you"  )
    for (i <-list){
          println(i)
    }

    var start = 1
    var stop =10
    for (i<-start to stop){
      println(i  + "第一种循环")
    }

    for (i<- start until stop){
      println(i + "第二种循环")
    }

    for (i<-start to stop if i==10){
       println(i)//10
    }
    
    for (i<-list if !i.startsWith("K") if i.contains("i")){
        println(i)
    }
  }
}

object Test4 {
  def main(args: Array[String]): Unit = {
    var start: Int = 1
    var stop = 10
    for (i<-start to stop ; j=9-i){
        println("引入变量" + j + i)
    }

    for (i<-start until stop  ; j<-1 to 9){
      println(j + "*" + i +"" + "=" + j*i  + "嵌套循环\r\n")
    }

    for (i<-Range(start,stop,2)){
      println(i + "Range 步长")
    }

    for (i<-start to stop by(2)){
      println(i + "步长")
    }

    var context = for (i<- start to stop) yield (
        if (i % 2 ==0)   i  else "不是偶数"
    )

    println(context)//Vector(不是偶数, 2, 不是偶数, 4, 不是偶数, 6, 不是偶数, 8, 不是偶数, 10)

  }
}

代码:
```
import util.control.Breaks._

object Test5 {
  def main(args: Array[String]): Unit = {
    //求和 ,到20的数是....
    var i = 0
    var sum = 100

    breakable {
      while (i < 10) {
        i += 1
        if (i > 20) {
          println(i)
          break()
        }
      }
    }
    sum = 100
    var lemp: Boolean = true
    for (num <- 10 to sum if lemp == true) {
      sum += num
      if (sum > 19999) {
        lemp = false
        println(sum)
      }
    }
    //continue功能实现  循环守卫 为真则执行
    for (num1 <- 1 to 10 if (num1 != 2 && num1 != 3)) {
      println(num1)
    }
  }
}
```
