**yield方法**
```
object Test8 {
  def main(args: Array[String]): Unit = {
    //map  yield 方法从现有map生成一个新的map
    var map1 = Map(1 -> 1, 2 -> 2, 3 -> 3, 4 -> 4)
    val map2 = for ((k, v) <- map1) yield (k, v - 1)
    println(map2)
    var map3 = for ((k, v) <- map1) yield (v, k)
    println(map3)
  }
}
```
