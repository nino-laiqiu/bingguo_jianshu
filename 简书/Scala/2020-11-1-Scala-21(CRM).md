package com.tongyongtao.scalaproject.CustomerView

import com.tongyongtao.scalaproject.Bean.User
import com.tongyongtao.scalaproject.Customerservice.Service

import scala.io.StdIn

object View {
  //提供一个展示功能
  def exhibition() = {
    println("*******  ********\t\n" +
      "欢迎来到用户登录平台\t\n" +
      "**  电话:00000001  ***\t\n" +
      "   地址:上海10000   \t\n")
    //调用菜单功能
    menu()
  }

  //提供一个菜单功能
  def menu() = {
    //使用模式匹配
    //控制循环
    var loop = true
    //记录次数,超过5次就跳出循环
    var count: Int = 0
    do {
      println("1.添加功能\t\n2.修改功能\t\n3.删除功能\t\n4.列表显示\t\n5.退出功能")
      val number = StdIn.readChar()
      number match {
        //跳转功能
        case '1' => add()
        case '2' =>
        case '3' => delete()
        case '4' => show()
        case '5' => loop = false
        case _ =>
          count += 1
          println("输入错误重新输入1,2,3,4,5")

      }
      if (count >= 5) {
        loop = false
      }
    }
    while (loop)
    println("退出")
  }

  //添加功能
  def add(): Unit = {
    //考虑把这些移动到service中
    println("输入您的属性")
    println("输入ID")
    var ID = StdIn.readLine()
    println("输入name")
    var name = StdIn.readLine()
    println("输入性别")
    var gender = StdIn.readChar()
    println("输入年龄")
    var age = StdIn.readShort()
    println("输入tel")
    var tel = StdIn.readLine()
    println("输入email")
    var email = StdIn.readLine()
    var user = new User(ID, name, gender, age, tel, email)
    Service.add(user)
    println("添加成功")
  }

  //删除功能
  def delete() = {
    println("输入您要删除用户的ID")
    val ID = StdIn.readLine()
    Service.delete(ID)
  }

  //展示功能
  def show() = {
    Service.show()
  }

}
