        hashSet4.add("天津");
        hashSet5.add("杭州");
        hashSet5.add("大连");
        //加入到map中
        bc.put("k1", hashSet1);
        bc.put("k2", hashSet2);
        bc.put("k3", hashSet3);
        bc.put("k4", hashSet4);
        bc.put("k5", hashSet5);
        //allAreas 用于存放所有地区
        HashSet<String> allAreas = new HashSet<>();
        allAreas.add("北京");
        allAreas.add("上海");
        allAreas.add("天津");
        allAreas.add("广州");
        allAreas.add("成都");
        allAreas.add("大连");
        allAreas.add("杭州");
        allAreas.add("深圳");
        //存放有选择的地区
        ArrayList<String> select = new ArrayList<String>();
        //定义一个临时集合,取交集
        HashSet<String> tempSet = new HashSet<String>();
        String maxkey = null;
        //定义一个maxkey 如果不为null 则加入到 select
        while (allAreas.size() != 0) {
            maxkey = null;
            for (String key : bc.keySet()) {
                tempSet.clear();
                HashSet<String> areas = bc.get(key);
                tempSet.addAll(areas);
                //求出tempSet 和allAreas 的交集 ,并把结果赋予给tempSet
                tempSet.retainAll(allAreas);
                if (tempSet.size() > 0 && (maxkey == null || tempSet.size() > bc.get(maxkey).size())) {
                    ;
                    maxkey = key;
                }
            }
            if (maxkey != null) {
                select.add(maxkey);
                allAreas.removeAll(bc.get(maxkey));
            }
        }
        System.out.println(select);
    }
}

public static void main(String[] args) {
        String a1 ="我不知道你在哪我知道";
        String b1 ="我知道";
        int i = boLi(a1, b1);
        System.out.println(i);
    }
    public static int boLi(String a , String b){
//将字符串型转换为字符型
        char[] st =  a.toCharArray();
        char[] st1 = b.toCharArray();
        int lon = st.length;
        int lon1 = st1.length;
        int i =0;
        int j =0;
//暴力匹配的核心
        while (i < lon && j< lon1){
            if (st[i] == st1[j]){
                i++;
                j++;
            }
            else {
                i = i + 1 -j;
                j =0;
            }
        }
//什么时候匹配成功
        if (j == lon1){
            return i-j;
        }
        else {
            return -1;
        }
    }
}
