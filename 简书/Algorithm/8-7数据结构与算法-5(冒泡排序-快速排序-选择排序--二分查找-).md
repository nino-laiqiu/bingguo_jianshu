4.快速排序代码实现(递归思想)
```
public class Demo8 {
    public static void main(String[] args) {


        int[] array = {1, 4, 6, 77, 77, 43};
        get2(array,0,array.length-1);
        System.out.println(Arrays.toString(array));
    }

    public static void get2(int[] array ,int low ,int hight){
        if (low > hight){
            return;
        }
        int i = low;
          int j = hight;
             int base =array[low];

             while (i < j){   
//上面的这个while的作用
                 while (array[j] >=base && i <j){//&&i<j 必须要加上去
                     j--;
                 }
                 while (array[i] <=base && i< j){
                     i++;
                 }
                 if (i < j) {
                     int temp = array[i];
                     array[i] = array[j];
                     array[j] = temp;
                 }
                 }
                 array[low] =array[i];
                 array[i] =base;


             get2(array,low,i-1);
             get2(array,i+1,hight);
    }
}
```
##对于交换的代码理解: 最终目的的array[i] 值放在左边  使用的中间值起传递作用(赋值)

    public static int get(int[] array,int get){
        int min =0;
        int max = array.length-1;
        int bl =(min + max)/2;
        while (min <= max){
            if (get <array[bl] ){
                max = bl -1;
            }
           else if (get > array[bl]){
                min =bl+1;
