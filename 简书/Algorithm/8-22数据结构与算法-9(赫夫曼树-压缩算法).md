```
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

public class Demo1 {
    //计数器
   static int count;
    public static void main(String[] args) {
        int[] arr = {1,3,4,8,5,7,25,10,81};
        //测试
        Node node = testhefumanshu(arr);
        proOrder(node);


    }
    //前序遍历的方法
    public  static void  proOrder(Node node){
        if (node != null){
            node.proOrder();
        }
        else {
            System.out.println("赫夫曼树为空");
        }
    }
    //构建一个赫夫曼树
    public  static  Node  testhefumanshu(int[] arr){
        //
        //

        List<Node> nodes = new ArrayList<>();
        //遍历这个数组,把元素放入集合中
        for (int value : arr) {
            //
            nodes.add(new Node(value));
        }
        //循环,直到最后只有一个根节点
        while (nodes.size() > 1) {

            count ++;
            //排序
            Collections.sort(nodes);

            //查看效果
            System.out.println("第" + count + "次排序" + nodes);

            //取出最小的两个节点
            Node leftnode = nodes.get(0);
            Node righrnode = nodes.get(1);

            //构建一个新节点
            Node newnode = new Node(leftnode.value + righrnode.value);
            newnode.left = leftnode;
            newnode.right = righrnode;

            //把最小的两个节点删除
            nodes.remove(leftnode);
            nodes.remove(righrnode);

            //把新的节点增添
            nodes.add(newnode);

            //重新排序
            // Collections.sort(nodes);
            //查看效果
            // System.out.println("第二次排序" + nodes);

            //发现规律
        }
        return  nodes.get(0);
    }
}
class Node implements Comparable<Node>{
    //实现接口  排序
    int value;
    Node left;
    Node right;
//构造方法
    public Node(int value) {
        this.value = value;
    }

    @Override
    public int compareTo(Node node) {
        //从小到大排序
        return this.value -node.value;
    }
    //复习二叉树的遍历方法:前序遍历 中序遍历 后序遍历
    public  void  proOrder(){
        System.out.println(this);
        if (this.left !=null){
            this.left.proOrder();
        }
        if (this.right!=null){
            this.right.proOrder();
        }
    }
    //重写tostring方法

    @Override
    public String toString() {
        return "Node{" +
                "value=" + value +
                '}';
    }
}
```
