//管理EmpLinkedList
class MyHashTab {
    public EmpLinkedList[] empLinkedList;
    public int size;//管理多少个List
    //需要一个构造器初始化

    public MyHashTab(int size) {
        this.size = size;
        empLinkedList = new EmpLinkedList[size];
        //这里仅仅初始化了数组 链表还没有初始化...
        for (int i = 0; i < size; i++) {
            empLinkedList[i] = new EmpLinkedList();
        }

    }

    //emp要添加到那个list中
    public int orientationFunction(int id) {
        return id % size;
    }

    //添加数据
    public void add(Emp emp) {
        //要添加到哪一个list
        int i = orientationFunction(Integer.parseInt(emp.id));
        empLinkedList[i].add(emp);
    }

    //遍历
    public void list() {
        for (int i = 0; i < size; i++) {
            empLinkedList[i].list(i);
        }
    }

    //找到指定id的职工信息  传入id
    public void findEmpbyid(String id) {
        //首先判断这个id在哪一个链表中
        int i = orientationFunction(Integer.parseInt(id));
        Emp emp = empLinkedList[i].findEmpbyid(id);
        if (emp == null) {
            System.out.println("没找到");
        } else {
            System.out.println(id + "在第" + i + "号链表上,信息为" + emp);
        }
    }

    public void deleteEmp(String id) {
        int i = orientationFunction(Integer.parseInt(id));
        empLinkedList[i].deleteEmp(id);
    }
}

class EmpLinkedList {
    Emp head;

    //假设添加都是添加到尾部 添加数据
    public void add(Emp emp) {
        //如果是添加一个员工
        if (head == null) {
            System.out.println("这是这条链表第一个员工");
            head = emp;
            return;
        }
        //不是第一个员工
        //辅助节点
        Emp tmpEmp = head;
        while (true) {
            if (tmpEmp.next == null) {
                //后面没有数据了 要添加了
                break;
            }
            //否则后移
            tmpEmp = tmpEmp.next;
        }
        //退出循环后 把emp添加到尾部
        tmpEmp.next = emp;
    }

    public void list(int i) {
        if (head == null) {
            System.out.println("第" + i + "条链表没有数据");
            return;
        }
        //辅助节点
        Emp tmpEmp = head;
        while (true) {
            System.out.println(tmpEmp);
            //说明到最后了 ,否则后移
            if (tmpEmp.next == null) {
                System.out.println("遍历到最后了");
                break;
            }
            tmpEmp = tmpEmp.next;
        }
    }

    //查找id是否在table中返回Emp
    public Emp findEmpbyid(String id) {
        if (head == null) {
            System.out.println("此节点为空");
            return null;
        }
        //否则 辅助节点
        Emp tmpEmp = head;
        while (true) {
            if (tmpEmp.id.equals(id)) {
                System.out.println("找到");
                break;
            }
            //tmpEmp = tmpEmp.next;....出错
            //如果没有找到
            if (tmpEmp.next == null) {
                //把这个节点重置,否则返回的右信息的这个节点
                tmpEmp = null;
                break;
            }
            //否则  ....注意这条语句的顺序
            tmpEmp = tmpEmp.next;
        }

        return tmpEmp;
    }

    //删除节点信息  不能删除第一个节点......
    public Emp deleteEmp(String id) {
        //首先判断首节点
        if (head == null) {
            System.out.println("该节点没有信息");
            return null;
        }
        //辅助节点
        Emp tmpEmop = head;
        //因为可能有重复节点,所以要遍历完
        while (true) {
            if (tmpEmop.next.id.equals(id)) {
                //删除这个节点,就是把这个节点指向下一个节点
                System.out.println("找到了");
                //注意....
                tmpEmop.next = tmpEmop.next.next;
                break;
                //删除tmpEmop.next节点信息
            }
            if (tmpEmop.next == null) {
                //遍历完成,找到返回,找不到 ,
                System.out.println("遍历结束");
                break;
            }
            tmpEmop = tmpEmop.next;
        }
        return null;
    }
}

class Emp {
    public String id;
    public String name;
    //指针指向下一个节点
    public Emp next;

    public Emp() {
    }

    public Emp(String id, String name) {
        this.id = id;
        this.name = name;
    }

    @Override
    public String toString() {
        return "Emp{" +
                "id='" + id + '\'' +
                ", name='" + name + '\'' +
                '}';
    }
}
