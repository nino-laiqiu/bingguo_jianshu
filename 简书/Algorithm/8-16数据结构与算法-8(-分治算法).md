#代码实现
```
public class Demo6 {
    static  int time;
    public static void main(String[] args) {

        hanluota(40, 'A', 'B', 'C');
    }
    public static void move(int desk, char M, char N) {
        System.out.println("把第" + desk + "盘子从" + M + "----->" + N  + "一共跑了" + (time++)+ "次");
    }
    public static void hanluota(int num, char a, char b, char c) {
        if (num == 1) {
            move(num, a, c);
        } else {
            //把n-1个盘子移动到b
            hanluota(num - 1, a, c, b);
            //把A盘中最后一个盘子移动到C
            move(num, a, c);
            //把b中的n-1个盘子移动到C盘中
            hanluota(num - 1, b, a, c);
            //两个递归中间的量理解为一个介质
        }
    }
}
```
