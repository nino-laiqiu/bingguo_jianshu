    val temp = new Array[Int](array.length)
    mergesort(array, 0, array.length - 1, temp)

  //分与合
  def mergesort(array: Array[Int], left: Int, right: Int, temp: Array[Int]): Unit = {
    //注意这里不能 = left <= right
    //什么时候发生合并操作
    if (left < right) {
      var mid: Int = (left + right) / 2
      //向左分解
      mergesort(array, left, mid, temp)
      //向右分解
      mergesort(array, mid + 1, right, temp)

      //合并
      println("******")
      merge(array, left, mid, right, temp)
    }
  }

  //mid为中间的索引
  //合
  def merge(array: Array[Int], left: Int, mid: Int, right: Int, temp: Array[Int]): Unit = {
    var i = left;
    var j = mid + 1;
    var t: Int = 0
    //首先排序,比较两路的值,较小的填入temp数组中
    while (i <= mid && j <= right) {
      if (array(i) <= array(j)) {
        temp(t) = array(i)
        t += 1
        i += 1
      }
      else {
        temp(t) = array(j)
        j += 1
        t += 1
      }
    }

    //填充左边或者右边仍然有多余的元素
    while (i <= mid) {
      temp(t) = array(i)
      t += 1
      i += 1
    }
    while (j <= right) {
      temp(t) = array(j)
      t += 1
      j += 1

    }
    //把temp数组中的元素复制到原来的数组
    //为什么从0开始?拷贝数组从0开始
    t = 0
    var templeft = left
    println(templeft + "    " + right)
    while (templeft <= right) {
      array(templeft) = temp(t)
      templeft += 1
      t += 1
    }
  }
}
