>/etc/sysconfig/network-scripts/ifcfg-ens33

```

UUID 删除
```

#2.JAVA_HOME的配置
解压到指定的目录中

>vi /etc/profile (G 到最后一行 o 插入到下一行)

>export     JAVA_HOME=/opt/java/jdk1.8.0_261


>export     PATH=$PATH:$JAVA_HOME/bin

>source /etc/profile (打开服务)

>java -version 测试是否完成

#4.MYSQL安装(提供一种安装方法)   (rpm  和 yum 都可以建议采用yum)
下载wget

>yum -y install wget

下载mysql安装包

>wget https://dev.mysql.com/get/mysql57-community-release-el7-8.noarch.rpm

安装MySQL

>yum -y install  mysql57-community-release-el7-8.noarch.rpm

安装mysql服务

>cd /etc/yum.repos.d/   (进入路径)

>yum -y install mysql-server (安装)

启动MySQL

>systemctl start mysqld

查看原始密码

>grep "password" /var/log/mysqld.log   

登录 

>mysql -uroot -pWYEc2Ir

>mysql> set global validate_password_policy=0;
>mysql> set global validate_password_length=1;   这个两个设置以后 密码很简单不会报错

修改密码

> ALTER USER 'root'@'localhost' IDENTIFIED BY '123456';

#6.ssh无密连接
安装ssh

>yum list | grep  ssh 

>yum -y install   文件

>ssh-keygen  回车(在每台机子上执行并且只需要执行一次就行了)

>ssh-copy-id  Linux01 (将自己的密钥给linux01 可执行多次)

**查看本机公钥**

>cd ~/.ssh  (进入目录)
>cat id_rsa.pub

>cat authorized_keys (查看连接的其他机子的密钥)
