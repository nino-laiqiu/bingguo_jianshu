建造者模式（Builder Pattern）使用多个简单的对象一步一步构建成一个复杂的对象。这种类型的设计模式属于创建型模式，它提供了一种创建对象的最佳方式


##2.代码示例
```
public class Client {
    public static void main(String[] args) {
        PersonDirector personDirector = new PersonDirector();
        Person createperson = personDirector.createperson(new ManBuilder());
        System.out.println(createperson.getFoot());
        System.out.println(createperson);
    }
}

//具体的产品
class Person {
    private String head;
    private String budy;
    private String foot;

    public Person() {

    }

    @Override
    public String toString() {
        return "Person{" +
                "head='" + head + '\'' +
                ", budy='" + budy + '\'' +
                ", foot='" + foot + '\'' +
                '}';
    }

    public Person(String head, String budy, String foot) {
        this.head = head;
        this.budy = budy;
        this.foot = foot;
    }

    public String getHead() {
        return head;
    }

    public void setHead(String head) {
        this.head = head;
    }

    public String getBudy() {
        return budy;
    }

    public void setBudy(String budy) {
        this.budy = budy;
    }

    public String getFoot() {
        return foot;
    }

    public void setFoot(String foot) {
        this.foot = foot;
    }
}

//建造者
interface Bulder {
    void buildhead();

    void buildbudy();

    void buildfoot();

    Person buildperson();
}

//具体的实现
class ManBuilder implements Bulder {
    Person person;

    public ManBuilder() {
        this.person = new Person();
    }

    @Override
    public void buildhead() {
        person.setHead("头");
    }

    @Override
    public void buildbudy() {
        person.setBudy("身体");
    }

    @Override
    public void buildfoot() {
        person.setFoot("脚");
    }

    @Override
    public Person buildperson() {
        return person;
    }
}

class PersonDirector {
    public Person createperson(Bulder builder) {
        builder.buildhead();
        builder.buildbudy();
        builder.buildfoot();
        return builder.buildperson();
    }
}
```
