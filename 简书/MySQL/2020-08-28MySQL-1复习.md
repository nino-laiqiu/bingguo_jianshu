5.基础命令
SHOW DATABASES;----查看所有数据库
CREATE DATABASE 表名;---建数据库
对建库和建表的规范 :db_  tb_
USE 库名;----进入某一个库
SHOW TABLES;---展示某一个库的所有表
CREATE TABLE 表名();----简单建表
DROP 库/表  删除
对表的操作:(大写)
insert into 表名()  value();----简单增加表的数据
delete from 表名 where ;----删除
update from 表名 set ..=.. where....-----更新表内数据
select * from 表名---查看表中数据
desc 表名 -----查看表的字段
